def MSE(y, y_predict):
    return np.average(
        np.array(list(map(lambda x: x**2, np.subtract(y, y_predict)))))
In [12]:

lambda_errs = []
for lam in lambda_set:
    error_for_lam = 0
    for i in range(5):
        folded_X_train = \
            np.concatenate((X_train[:i * pivot], X_train[(i + 1) * pivot:]))
        folded_Y_train = \
            np.concatenate((Y_train[:i * pivot], Y_train[(i + 1) * pivot:]))
        folded_X_test = X_train[i*pivot:(i+1)*pivot]
        folded_Y_test = Y_train[i*pivot:(i+1)*pivot]
        w = np.matmul(
            np.linalg.inv(np.matmul(np.transpose(folded_X_train), folded_X_train) +
                                    lam * np.identity(D)),
            np.matmul(np.transpose(folded_X_train), folded_Y_train))
        y_predict = np.matmul(folded_X_test, w)
        error_for_lam += SSE(folded_Y_test, y_predict)
    error_for_lam /= K
    lambda_errs.append(error_for_lam)
